#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <vector> 

std::vector<std::string> splitString(std::string& str, std::string delimit){
    std::cout << "Started splitString ..." << std::endl;
    
    std::vector<std::string> result;
    if (str.size() != 0){
        int prevIdx = 0;
        int curIdx = 0;
        std::string buffer;
        while (curIdx != std::string::npos){
            curIdx = str.find(delimit, prevIdx);
            buffer = str.substr(prevIdx, curIdx-prevIdx);
            std::cout << buffer << std::endl; 
            prevIdx = curIdx + delimit.size();
            result.push_back(buffer);
        }
    }
    
    std::cout << "... Done splitString" << std::endl;
    return result;
}

void parseNode(){
    
    std::fstream file;
    file.open("text.txt", std::ios::in);
    
    std::string line;
    while (std::getline(file, line)){
        
        for (auto ele : splitString(line, ";")){
            std::cout << ele << std::endl; 
        }
        
    }
    
    
}


int main()
{
    parseNode();


    return 0;
}
